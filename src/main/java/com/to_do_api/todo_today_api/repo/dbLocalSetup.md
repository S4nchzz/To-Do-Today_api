CREATE DATABASE TO_DO_TODAY;
USE TO_DO_TODAY;

CREATE TABLE USER (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    USERNAME VARCHAR(20) NOT NULL,
    SALT VARCHAR(16) NOT NULL,
    PASS BINARY(32) NOT NULL,
    EMAIL VARCHAR(50) NOT NULL,
    LOGGED_IN BOOLEAN NOT NULL,
    CONSTRAINT uniqueName UNIQUE (username),
    CONSTRAINT uniqueEmail UNIQUE (email)
);

CREATE TABLE USER_PROFILE (
    ID INT PRIMARY KEY,
    PROFILE_PICTURE MEDIUMBLOB,
    PHONE VARCHAR(50),
    FOREIGN KEY (ID) REFERENCES USER(ID)
);

CREATE TABLE USER_TEMPORAL_TOKEN (
    TOKEN VARCHAR(255) NOT NULL,
    USERID INT,
    FOREIGN KEY (USERID) REFERENCES USER(ID)
);

CREATE TABLE KEEP_LOGGED_TOKENS(
    TOKEN VARCHAR(255) NOT NULL,
    USERID INT,
    FOREIGN KEY (USERID) REFERENCES USER(ID)
);

CREATE TABLE AUTH_API_TOKENS (
    ID INTEGER PRIMARY KEY AUTO_INCREMENT,
    TOKEN VARCHAR(255) NOT NULL
);

CREATE TABLE TEAMS(
    TEAM_KEY VARCHAR(5) PRIMARY KEY,
    TITLE VARCHAR(30),
    DESCRIPTION VARCHAR(255),
	 ADMINISTRATOR INTEGER NOT NULL,
    DATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (ADMINISTRATOR) REFERENCES USER(ID)
);


CREATE TABLE TODO (
    ID INTEGER PRIMARY KEY AUTO_INCREMENT,
    USERID INT,
    HEADER VARCHAR(255) NOT NULL,
    CONTENT VARCHAR(255) NOT NULL,
    DATE VARCHAR(255) NOT NULL,
    FAV BOOLEAN,
    ENDED BOOLEAN NOT NULL,
    TEAM VARCHAR(5),
    FOREIGN KEY (TEAM) REFERENCES TEAMS(TEAM_KEY),
    FOREIGN KEY (USERID) REFERENCES USER(ID)
);
